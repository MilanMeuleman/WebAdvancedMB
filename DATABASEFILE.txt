CREATE DATABASE events;

USE events;

CREATE TABLE Persoon(
PersoonNaam VARCHAR(50),
PersoonID INT(8) NOT NULL,
PRIMARY KEY(PersoonID)
);

CREATE TABLE Afspraak(
EventID	 INT(8) NOT NULL,
PersoonID INT(8) NOT NULL,
StartDatum DATE,
EindDatum DATE,
beschrijving VARCHAR(150),
PRIMARY KEY(EventID),
FOREIGN KEY(PersoonID) REFERENCES Persoon(PersoonID)
);
